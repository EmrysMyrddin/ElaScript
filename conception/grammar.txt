//Rules
Script ::= Statement+

Statement ::= Command | Parallelized

Parallelized ::= SPLIT Body JOIN

Body ::= BodyPart (PARALLELSEPARATOR BodyPart)+

BodyPart ::= Statement+

Command ::= Function LP ParamList RP SEQUENCIALSEPARATOR

Function ::= LETTER (LETTER | NUMBER)*

ParamList ::= Param ( COMMA Param )*

Param ::= NUMBER+

//Lexems
LP ::= '('
RP ::= ')'
LETTER ::= [a-zA-Z]
NUMBER ::= [0-9]
TIMER ::= 'wait'
SEQUENCIALSEPARATOR ::= ';'
PARALLELSEPARATOR ::= '||'
COMMA ::= ','
SPLIT ::= '['
JOIN ::= ']'
