//Rules
Script ::= Statement+

Statement ::= Command | Parallelized

Parallelized ::= split (Statement+ parallelSeparator Statement+)+ join

Command ::= Function lp ParamList rp sequentialSeparator

Function ::= letter (letter | number)*

ParamList ::= Param ( comma Param )*

Param ::= number+

//Lexems
lp ::= '('
rp ::= ')'
letter ::= [a-zA-Z]
number ::= [0-9]
timer ::= 'wait'
sequentialSeparator ::= ';'
parallelSeparator ::= '||'
comma ::= ','
split ::= '['
join ::= ']'
